/* Interface: HR_employeeGrid */
/* Type: Report / Dashboard component */
/* Description: Searchable, filterable grid of employees with inline actions */
/* Rule Inputs: ri!department (Text, optional filter) */

a!localVariables(
  local!searchText: "",
  local!selectedDepartment: ri!department,
  local!selectedStatus: "ACTIVE",
  local!pagingInfo: a!pagingInfo(
    startIndex: 1,
    batchSize: 25,
    sort: a!sortInfo(
      field: recordType!Employee.fields.lastName,
      ascending: true
    )
  ),
  local!employees: a!queryRecordType(
    recordType: recordType!Employee,
    fields: {
      recordType!Employee.fields.id,
      recordType!Employee.fields.firstName,
      recordType!Employee.fields.lastName,
      recordType!Employee.fields.email,
      recordType!Employee.fields.department,
      recordType!Employee.fields.status,
      recordType!Employee.fields.startDate,
      recordType!Employee.relationships.manager.fields.firstName
    },
    filters: a!queryLogicalExpression(
      operator: "AND",
      filters: {
        a!queryFilter(
          field: recordType!Employee.fields.department,
          operator: "=",
          value: local!selectedDepartment
        ),
        a!queryFilter(
          field: recordType!Employee.fields.status,
          operator: "=",
          value: local!selectedStatus
        ),
        a!queryFilter(
          field: recordType!Employee.fields.lastName,
          operator: "includes",
          value: local!searchText
        )
      },
      ignoreFiltersWithEmptyValues: true
    ),
    pagingInfo: local!pagingInfo
  ),

  {
    a!sectionLayout(
      label: "Filters",
      contents: {
        a!columnsLayout(
          columns: {
            a!columnLayout(
              contents: {
                a!textField(
                  label: "Search by Last Name",
                  value: local!searchText,
                  saveInto: {
                    local!searchText,
                    a!save(local!pagingInfo.startIndex, 1)
                  },
                  refreshAfter: "KEYPRESS",
                  placeholder: "Type to search..."
                )
              }
            ),
            a!columnLayout(
              contents: {
                a!dropdownField(
                  label: "Department",
                  placeholder: "All Departments",
                  choiceLabels: cons!HR_DEPARTMENT_LABELS,
                  choiceValues: cons!HR_DEPARTMENT_VALUES,
                  value: local!selectedDepartment,
                  saveInto: {
                    local!selectedDepartment,
                    a!save(local!pagingInfo.startIndex, 1)
                  }
                )
              }
            ),
            a!columnLayout(
              contents: {
                a!radioButtonField(
                  label: "Status",
                  choiceLabels: {"Active", "Inactive", "All"},
                  choiceValues: {"ACTIVE", "INACTIVE", ""},
                  value: local!selectedStatus,
                  saveInto: {
                    local!selectedStatus,
                    a!save(local!pagingInfo.startIndex, 1)
                  },
                  choiceLayout: "COMPACT"
                )
              }
            )
          }
        )
      }
    ),
    a!gridField(
      label: "Employees (" & local!employees.totalCount & ")",
      data: local!employees,
      columns: {
        a!gridColumn(
          label: "Name",
          sortField: recordType!Employee.fields.lastName,
          value: a!linkField(
            links: a!recordLink(
              label: fv!row[recordType!Employee.fields.lastName] & ", " &
                     fv!row[recordType!Employee.fields.firstName],
              recordType: recordType!Employee,
              identifier: fv!row[recordType!Employee.fields.id]
            )
          )
        ),
        a!gridColumn(
          label: "Email",
          sortField: recordType!Employee.fields.email,
          value: fv!row[recordType!Employee.fields.email]
        ),
        a!gridColumn(
          label: "Department",
          sortField: recordType!Employee.fields.department,
          value: fv!row[recordType!Employee.fields.department]
        ),
        a!gridColumn(
          label: "Manager",
          value: fv!row[recordType!Employee.relationships.manager.fields.firstName]
        ),
        a!gridColumn(
          label: "Start Date",
          sortField: recordType!Employee.fields.startDate,
          value: text(fv!row[recordType!Employee.fields.startDate], "MMM dd, yyyy"),
          align: "END"
        ),
        a!gridColumn(
          label: "Status",
          sortField: recordType!Employee.fields.status,
          value: a!tagField(
            tags: a!tagItem(
              text: fv!row[recordType!Employee.fields.status],
              backgroundColor: if(
                fv!row[recordType!Employee.fields.status] = "ACTIVE",
                "POSITIVE",
                "NEGATIVE"
              )
            )
          ),
          align: "CENTER",
          width: "NARROW"
        )
      },
      pagingSaveInto: local!pagingInfo,
      rowHeader: 1,
      showExportButton: true,
      emptyGridMessage: "No employees found matching your criteria."
    )
  }
)

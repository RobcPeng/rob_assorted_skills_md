/* Rule: rule!HR_calculateTenure */
/* Description: Calculates human-readable tenure from a start date to today */
/* Inputs: ri!startDate (Date) */
/* Returns: Text (e.g., "2 years, 3 months" or "15 days") */

a!localVariables(
  local!days: tointeger(today() - ri!startDate),
  local!years: floor(local!days / 365.25),
  local!remainingDays: local!days - tointeger(local!years * 365.25),
  local!months: floor(local!remainingDays / 30.44),
  local!finalDays: local!remainingDays - tointeger(local!months * 30.44),

  if(
    a!isNullOrEmpty(ri!startDate),
    "N/A",
    if(
      local!years > 0,
      text(local!years, "0") & if(local!years = 1, " year", " years")
        & if(
            local!months > 0,
            ", " & text(local!months, "0") & if(local!months = 1, " month", " months"),
            ""
          ),
      if(
        local!months > 0,
        text(local!months, "0") & if(local!months = 1, " month", " months"),
        text(local!finalDays, "0") & if(local!finalDays = 1, " day", " days")
      )
    )
  )
)
